# tests/7-base_geometry.txt
BaseGeometry = __import__('7-base_geometry').BaseGeometry
bg = BaseGeometry()

# Test attributes
hasattr(bg, 'area')
hasattr(bg, 'integer_validator')

# Test area() raises Exception
try:
    bg.area()
except Exception as e:
    print(e)

# Test integer_validator() with valid integers
bg.integer_validator("my_int", 12)
bg.integer_validator("width", 89)

# Test integer_validator() with invalid types
try:
    bg.integer_validator("name", "John")
except Exception as e:
    print("[{}] {}".format(e.__class__.__name__, e))

try:
    bg.integer_validator("age", 3.5)
except Exception as e:
    print("[{}] {}".format(e.__class__.__name__, e))

try:
    bg.integer_validator("age", True)
except Exception as e:
    print("[{}] {}".format(e.__class__.__name__, e))

    bg.integer_validator("age", (4,))

    print("[{}] {}".format(e.__class__.__name__, e))

    bg.integer_validator("age", [3])

    print("[{}] {}".format(e.__class__.__name__, e))

try:
    bg.integer_validator("age", {3, 4})
except Exception as e:
    print("[{}] {}".format(e.__class__.__name__, e))

try:
    bg.integer_validator("age", None)
except Exception as e:
    print("[{}] {}".format(e.__class__.__name__, e))

# Test integer_validator() with non-positive integers
